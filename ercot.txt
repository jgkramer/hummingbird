  function headerDimensions(debug = false) {
    // find the dimensions of each drop-down and center them
    screenw = window.innerWidth || document.documentElement.clientWidth || document.body.clientWidth;
    if (screenw <= 1200) return; // don't bother on mobile

    let leftLimit = Math.round($("header").offset().left);

    let $menuLi = $('header.combined nav > ul > li:visible');

      const spotlightWidth = 280; // widths must match CSS

      let $this = $(this);
      let colCount = $this.find('.dropdown > ul').length;  // columns
      let sptCount = $this.find('.dropdown > div').length; // spotlights
      let itemWidth = Math.round(colCount*columnWidth + sptCount*spotlightWidth);

      let right = -offset; // right-align with top nav
      let left = itemRight-right-itemWidth-itemLeft; // left position for full width

      let checkpoint = itemLeft+left;
      let adjustment;
      if (checkpoint < leftLimit)  // part of nav is hidden
        adjustment = itemLeft - leftLimit + left;
      else
        adjustment = 0;

      let adjRight = right+adjustment;
